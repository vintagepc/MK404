# This script just checks that the printer boots by looking for 'start' on the serial line.
ScriptHost::SetTimeoutMs(10000)
ScriptHost::SetQuitOnTimeout(1)
Serial0::WaitForLine(READY)
3DView::Snapshot(tests/snaps/GFXLite01)
Encoder::Press()
Serial0::WaitForLineContains(BED)
3DView::Snapshot(tests/snaps/GFXLite02)
3DVisuals::ToggleNozzleCam()
3DView::Snapshot(tests/snaps/GFXLite03)
3DVisuals::ToggleNozzleCam()
Board::Reset()
Board::WaitMs(100)
3DView::Snapshot(tests/snaps/GFXLite04)
Board::Quit()
